# Turns the rewriting engine on
RewriteEngine On 

# Prevents directory listing
Options All -Indexes

# Prevents access to .htaccess itself
<Files .htaccess>
 order allow,deny
 deny from all
</Files>

# My own 404 page
ErrorDocument 404 /404.html

# RSS flux
RewriteRule ^rss$               rss.xml [L]

# Last revision when empty
RewriteRule ^$                  -1.html
RewriteRule ^-1$                -1.html

# Explicit revision
RewriteRule ^([a-zA-Z0-9_]+)$   $1.html

# For PHP examples
php_flag magic_quotes_gpc Off
php_flag magic_quotes_runtime Off
php_flag magic_quotes_sybase Off
